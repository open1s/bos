### Locked Decisions (from CONTEXT.md)

**Trait-based dispatch** — `RpcHandler` trait:
```rust
#[async_trait]
pub trait RpcHandler: Send + Sync {
    async fn handle(&self, method: &str, payload: &[u8]) -> Result<Vec<u8>, RpcServiceError>;
}
```

**Service announces** — peer-to-peer discovery, no central registry
**Discovery topic** — `/rpc/discover/{service_name}`
**Discovery response** — `{ "topic_prefix": "/rpc/{service}" }`

### RpcService Design
- Wraps QueryableWrapper internally
- Deserializes incoming requests using JsonCodec
- Routes to handler method based on request metadata
- Responds with RpcResponse envelope (Ok or Err)
- Uses `async-trait` since RpcHandler methods are async

### RpcDiscovery Design
- Creates QueryableWrapper on `/rpc/discover/{service_name}`
- Responds with `{ "topic_prefix": "/rpc/{service}" }`
- RpcDiscovery::discover() uses session.query() to find services
- Caches discovered endpoints (first response wins)
</context>

//! RPC service wrapper

use std::sync::Arc;
use std::marker::PhantomData;

use async_trait::async_trait;
use tokio::task::JoinHandle;

use crate::queryable::QueryableWrapper;
use crate::rpc::error::RpcServiceError;
use crate::rpc::types::RpcResponse;
use crate::Session;
use crate::JsonCodec;

/// Trait for RPC service handlers.
/// Services implement this trait to handle incoming RPC method calls.
#[async_trait]
pub trait RpcHandler: Send + Sync {
    /// Handle an RPC method call.
    /// 
    /// `method` — the method name (e.g., "add", "multiply")
    /// `payload` — raw request bytes (caller's serialized request)
    /// 
    /// Returns raw bytes of the response (serialized by caller).
    async fn handle(&self, method: &str, payload: &[u8]) -> Result<Vec<u8>, RpcServiceError>;
}

/// Internal request envelope for method routing.
/// The caller sends: RpcRequest { method, payload }
#[derive(Debug, serde::Deserialize)]
pub struct RpcRequest {
    pub method: String,
    pub payload: Vec<u8>,
}

/// Internal response envelope for service responses.
/// The service returns: RpcResponse<T>
#[derive(Debug, Clone, serde::Serialize)]
pub struct RpcResponseEnvelope {
    status: String,
    #[serde(skip_serializing_if = "Option::is_none")]
    ok: Option<serde_json::Value>,
    #[serde(skip_serializing_if = "Option::is_none")]
    err: Option<RpcErrBody>,
}

#[derive(Debug, Clone, serde::Serialize)]
pub struct RpcErrBody {
    code: u32,
    message: String,
}

/// RpcService builder for fluent configuration.
#[derive(Debug, Clone, Default)]
pub struct RpcServiceBuilder {
    service_name: Option<String>,
    topic_prefix: Option<String>,
}

impl RpcServiceBuilder {
    pub fn new() -> Self {
        Self::default()
    }

    pub fn service_name(mut self, name: impl Into<String>) -> Self {
        self.service_name = Some(name.into());
        self
    }

    pub fn topic_prefix(mut self, prefix: impl Into<String>) -> Self {
        self.topic_prefix = Some(prefix.into());
        self
    }

    pub fn build(self) -> Result<RpcServiceUninit, RpcServiceError> {
        let service_name = self.service_name.ok_or_else(|| {
            RpcServiceError::Internal("service_name required".to_string())
        })?;
        let topic_prefix = self.topic_prefix.unwrap_or_else(|| format!("/rpc/{}", service_name));
        Ok(RpcServiceUninit {
            service_name,
            topic_prefix,
        })
    }
}

/// Uninitialized RpcService. Call .init() with a session.
#[derive(Debug)]
pub struct RpcServiceUninit {
    service_name: String,
    topic_prefix: String,
}

impl RpcServiceUninit {
    pub async fn init<H>(self, session: Arc<Session>, handler: Arc<H>) -> Result<RpcService, RpcServiceError>
    where
        H: RpcHandler,
    {
        let topic = format!("{}/dispatch", self.topic_prefix);
        RpcService::new(topic, handler).await
    }
}

/// RpcService — handles RPC requests via QueryableWrapper.
/// 
/// # Example
/// ```rust,ignore
/// struct CalcHandler;
/// 
/// #[async_trait]
/// impl RpcHandler for CalcHandler {
///     async fn handle(&self, method: &str, payload: &[u8]) -> Result<Vec<u8>, RpcServiceError> {
///         match method {
///             "add" => {
///                 let req: (i32, i32) = serde_json::from_slice(payload)?;
///                 Ok(serde_json::to_vec(&(req.0 + req.1))?)
///             }
///             _ => Err(RpcServiceError::Business {
///                 code: 404,
///                 message: format!("Unknown method: {}", method),
///             }),
///         }
///     }
/// }
/// 
/// let service = RpcServiceBuilder::new()
///     .service_name("calculator")
///     .build()
///     .unwrap();
/// service.init(session, Arc::new(CalcHandler)).await?;
/// service.into_task()?;
/// ```
pub struct RpcService {
    topic: String,
    queryable: QueryableWrapper<RpcRequest, RpcResponseEnvelope>,
    handler: Arc<dyn RpcHandler>,
    codec: JsonCodec,
}

impl RpcService {
    pub async fn new(topic: String, handler: Arc<dyn RpcHandler>) -> Result<Self, RpcServiceError> {
        let codec = JsonCodec;
        let queryable = QueryableWrapper::<RpcRequest, RpcResponseEnvelope>::new(topic.clone())
            .with_handler(move |request: RpcRequest| {
                let handler = handler.clone();
                let codec = codec;
                async move {
                    // Call the handler
                    let result = handler.handle(&request.method, &request.payload).await;
                    // Wrap in envelope
                    match result {
                        Ok(response_bytes) => {
                            // The response bytes are the inner value - wrap as Ok
                            let value: serde_json::Value = serde_json::from_slice(&response_bytes)
                                .unwrap_or(serde_json::Value::Null);
                            Ok(RpcResponseEnvelope {
                                status: "ok".to_string(),
                                ok: Some(value),
                                err: None,
                            })
                        }
                        Err(e) => {
                            Ok(RpcResponseEnvelope {
                                status: "err".to_string(),
                                ok: None,
                                err: Some(RpcErrBody {
                                    code: match &e {
                                        RpcServiceError::Business { code, .. } => *code,
                                        RpcServiceError::Internal(_) => 500,
                                    },
                                    message: e.to_string(),
                                }),
                            })
                        }
                    }
                }
            });
        
        Ok(Self {
            topic,
            queryable,
            handler,
            codec,
        })
    }

    /// Initialize the queryable with a session.
    pub async fn init(&mut self, session: Arc<Session>) -> Result<(), RpcServiceError> {
        self.queryable.init(session).await.map_err(|e| {
            RpcServiceError::Internal(format!("Failed to init queryable: {}", e))
        })
    }

    /// Start the service as a background task.
    pub fn into_task(mut self) -> Result<JoinHandle<Result<(), crate::error::ZenohError>>, RpcServiceError> {
        self.queryable.into_task().map_err(|e| {
            RpcServiceError::Internal(format!("Failed to spawn task: {}", e))
        })
    }

    pub fn topic(&self) -> &str {
        &self.topic
    }
}

impl Clone for RpcService {
    fn clone(&self) -> Self {
        Self {
            topic: self.topic.clone(),
            queryable: self.queryable.clone(),
            handler: self.handler.clone(),
            codec: JsonCodec,
        }
    }
}

//! RPC service discovery

use std::sync::Arc;
use std::time::Duration;

use serde::{Deserialize, Serialize};

use crate::queryable::QueryableWrapper;
use crate::Session;
use crate::error::ZenohError;

/// Discovery response body.
#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct DiscoveryInfo {
    pub topic_prefix: String,
    pub service_name: String,
}

/// RpcDiscovery — peer-to-peer service discovery.
/// 
/// Services announce themselves on `/rpc/discover/{service_name}`.
/// Clients query this topic to find available services.
/// 
/// # Example
/// ```rust,ignore
/// // Service: announce itself
/// let discovery = RpcDiscovery::announce("calculator".to_string());
/// discovery.init(session.clone()).await?;
/// 
/// // Client: discover a service
/// let endpoints = RpcDiscovery::discover("calculator".to_string())
///     .session(session.clone())
///     .timeout(Duration::from_secs(5))
///     .query()
///     .await?;
/// ```
pub struct RpcDiscovery {
    service_name: String,
    topic: String,
    queryable: Option<QueryableWrapper<String, DiscoveryInfo>>,
    codec: crate::JsonCodec,
}

impl RpcDiscovery {
    /// Create a discovery announcer for a service.
    pub fn announce(service_name: impl Into<String>) -> Self {
        let service_name = service_name.into();
        Self {
            topic: format!("/rpc/discover/{}", service_name),
            service_name: service_name.clone(),
            queryable: None,
            codec: crate::JsonCodec,
        }
    }

    /// Initialize and start responding to discovery queries.
    pub async fn init(&mut self, session: Arc<Session>) -> Result<(), ZenohError> {
        let topic = self.topic.clone();
        let info = DiscoveryInfo {
            topic_prefix: format!("/rpc/{}", self.service_name),
            service_name: self.service_name.clone(),
        };
        
        let mut queryable = QueryableWrapper::new(topic)
            .with_handler(move |_query: String| Ok(info.clone()));
        
        queryable.init(session).await?;
        self.queryable = Some(queryable);
        Ok(())
    }

    /// Stop responding to discovery queries.
    pub fn shutdown(&mut self) {
        self.queryable = None;
    }

    /// Discovery query builder.
    pub fn discover(service_name: impl Into<String>) -> DiscoveryQueryBuilder {
        DiscoveryQueryBuilder {
            service_name: service_name.into(),
            session: None,
            timeout: Duration::from_secs(5),
        }
    }
}

#[derive(Debug, Clone)]
pub struct DiscoveryQueryBuilder {
    service_name: String,
    session: Option<Arc<Session>>,
    timeout: Duration,
}

impl DiscoveryQueryBuilder {
    pub fn session(mut self, session: Arc<Session>) -> Self {
        self.session = Some(session);
        self
    }

    pub fn timeout(mut self, timeout: Duration) -> Self {
        self.timeout = timeout;
        self
    }

    /// Query for the service and return the first discovery response.
    pub async fn query(self) -> Result<DiscoveryInfo, ZenohError> {
        let session = self.session.ok_or(ZenohError::NotConnected)?;
        let topic = format!("/rpc/discover/{}", self.service_name);
        
        let replies = session
            .get(&topic)
            .payload("")
            .timeout(self.timeout)
            .await?;
        
        while let Ok(reply) = replies.recv_async().await {
            if let Ok(sample) = reply.result() {
                let bytes = sample.payload().to_bytes();
                if let Ok(info) = serde_json::from_slice::<DiscoveryInfo>(bytes.as_ref()) {
                    return Ok(info);
                }
            }
        }
        
        Err(ZenohError::Query(format!(
            "No discovery response for service '{}'",
            self.service_name
        )))
    }

    /// Query and return ALL discovery responses.
    pub async fn query_all(self) -> Result<Vec<DiscoveryInfo>, ZenohError> {
        let session = self.session.ok_or(ZenohError::NotConnected)?;
        let topic = format!("/rpc/discover/{}", self.service_name);
        
        let replies = session
            .get(&topic)
            .payload("")
            .timeout(self.timeout)
            .await?;
        
        let mut results = Vec::new();
        while let Ok(reply) = replies.recv_async().await {
            if let Ok(sample) = reply.result() {
                let bytes = sample.payload().to_bytes();
                if let Ok(info) = serde_json::from_slice::<DiscoveryInfo>(bytes.as_ref()) {
                    results.push(info);
                }
            }
        }
        
        if results.is_empty() {
            return Err(ZenohError::Query(format!(
                "No discovery responses for service '{}'",
                self.service_name
            )));
        }
        
        Ok(results)
    }
}

impl Clone for RpcDiscovery {
    fn clone(&self) -> Self {
        Self {
            service_name: self.service_name.clone(),
            topic: self.topic.clone(),
            queryable: None, // Clones don't carry queryable
            codec: crate::JsonCodec,
        }
    }
}

#[cfg(test)]
mod tests {
    use super::*;
    use std::sync::Arc;
    use std::time::Duration;
    use async_trait::async_trait;
    use crate::{SessionManager, ZenohConfig};

    struct TestServiceHandler;

    #[async_trait]
    impl RpcHandler for TestServiceHandler {
        async fn handle(&self, method: &str, payload: &[u8]) -> Result<Vec<u8>, RpcServiceError> {
            match method {
                "echo" => {
                    Ok(payload.to_vec())
                }
                "add" => {
                    let (a, b): (i32, i32) = serde_json::from_slice(payload)
                        .map_err(|e| RpcServiceError::Internal(e.to_string()))?;
                    serde_json::to_vec(&(a + b))
                        .map_err(|e| RpcServiceError::Internal(e.to_string()))
                }
                "error" => {
                    Err(RpcServiceError::Business {
                        code: 500,
                        message: "intentional error".to_string(),
                    })
                }
                _ => Err(RpcServiceError::Business {
                    code: 404,
                    message: format!("Unknown method: {}", method),
                }),
            }
        }
    }

    #[tokio::test(flavor = "multi_thread", worker_threads = 2)]
    async fn test_rpc_full_cycle() {
        // Setup session
        let config = ZenohConfig::default();
        let manager = SessionManager::new(config);
        let session = manager.connect().await.expect("connect");
        let session = Arc::new(session);

        // Start service
        let handler = Arc::new(TestServiceHandler);
        let mut service = RpcServiceBuilder::new()
            .service_name("test")
            .build()
            .expect("build");
        service.init(session.clone(), handler).await.expect("init");
        let _handle = service.into_task().expect("into_task");
        
        // Small delay for service to register
        tokio::time::sleep(Duration::from_millis(100)).await;

        // Start discovery (service announces)
        let mut discovery_announce = RpcDiscovery::announce("test");
        discovery_announce.init(session.clone()).await.expect("discovery init");

        // Client: discover service
        let info = RpcDiscovery::discover("test")
            .session(session.clone())
            .timeout(Duration::from_secs(5))
            .query()
            .await
            .expect("discovery query");
        assert_eq!(info.service_name, "test");
        assert_eq!(info.topic_prefix, "/rpc/test");

        // Client: call service via discovered endpoint
        let mut client = RpcClient::builder()
            .service("test")
            .method("echo")
            .timeout(Duration::from_secs(5))
            .build()
            .expect("build client");
        client.init(session.clone()).await.expect("client init");

        // Test echo
        let payload = serde_json::to_vec(&"hello").expect("serialize");
        let response: String = client.call(&payload).await.expect("call");
        assert_eq!(response, "hello");

        // Test add
        let mut add_client = RpcClient::builder()
            .service("test")
            .method("add")
            .timeout(Duration::from_secs(5))
            .build()
            .expect("build add client");
        add_client.init(session.clone()).await.expect("add init");
        let payload = serde_json::to_vec(&(10, 20)).expect("serialize add");
        let result: i32 = add_client.call(&payload).await.expect("add call");
        assert_eq!(result, 30);
    }
}
