# Phase 03 Plan 01: Scheduler - Workflow Execution Engine

## Objective

Implement a workflow scheduler that supports sequential, parallel, and conditional execution with configurable timeout and retry logic.

## Goals

1. **Workflow DSL** — Builder pattern for defining workflows
2. **Sequential Execution** — Execute steps in order, pass output as input
3. **Parallel Execution** — Run steps simultaneously, collect all results
4. **Conditional Branching** — Branch based on output pattern matching
5. **Step Timeout** — Configurable per-step timeout enforcement
6. **Retry Logic** — Exponential backoff with configurable max retries

```rust
pub struct WorkflowBuilder {

    description: String,
    steps: Vec<Step>,

    default_retries: u32,
    default_backoff: BackoffStrategy,

    pub fn new(name: String) -> Self {
        Self {
            name,
            description: String::new(),
            steps: Vec::new(),
            default_timeout: Duration::from_secs(60),
            default_retries: 3,
            default_backoff: BackoffStrategy::Exponential {
                base: Duration::from_millis(100),
                max: Duration::from_secs(10),
            },
        }
    }

    pub fn description(mut self, desc: String) -> Self {
        self.description = desc;
        self
    }

    pub fn default_timeout(mut self, timeout: Duration) -> Self {
        self.default_timeout = timeout;
        self
    }

    pub fn default_retries(mut self, retries: u32) -> Self {
        self.default_retries = retries;
        self
    }

    pub fn add_step(mut self, step: Step) -> Self {
        self.steps.push(step);
        self
    }

    /// Add sequential step (A → B → C)
    pub fn sequential_step(
        mut self,
        name: String,
        agent_id: Option<String>,
        input: serde_json::Value,
    ) -> Self {
        self.steps.push(Step {
            name,
            step_type: StepType::Sequential,
            agent_id,
            timeout: self.default_timeout,
            max_retries: self.default_retries,
            backoff: self.default_backoff.clone(),
        });
        self
    }

    /// Add parallel group (run A, B, C simultaneously)
    pub fn parallel_group(
        mut self,
        group_name: String,
        steps: Vec<(String, Option<String>, serde_json::Value)>,
    ) -> Self {
        // Steps will be marked with Parallel type
        for (name, agent_id, input) in steps {
            self.steps.push(Step {
                name,
                step_type: StepType::Parallel,
                agent_id,
                timeout: self.default_timeout,
                max_retries: self.default_retries,
                backoff: self.default_backoff.clone(),
            });
        }
        self
    }

    /// Add conditional branching
    pub fn branch(
        mut self,
        name: String,
        condition: ConditionType,
        true_branch: String,
        false_branch: String,
    ) -> Self {
        self.steps.push(Step {
            name,
            step_type: StepType::Conditional { condition },
            agent_id: None,  // Branching is local
            timeout: self.default_timeout,
            max_retries: 0,  // No retry for branching
            backoff: BackoffStrategy::Fixed { interval: Duration::ZERO },
        });
        self
    }

    pub fn build(self) -> Workflow {
        Workflow {
            name: self.name,
            description: self.description,
            steps: self.steps,
            default_timeout: self.default_timeout,
            default_retries: self.default_retries,
        }
    }
}
```

```rust
use super::BackoffStrategy;
use std::time::Duration;
use tokio::time::sleep;

/// Calculate delay for retry attempt
pub fn calculate_delay(strategy: &BackoffStrategy, attempt: u32, timeout: Duration) -> Duration {
    match strategy {
        BackoffStrategy::Exponential { base, max } => {
            let delay = base * 2u32.pow(attempt);
            if delay > *max {
                *max
            } else {
                delay
            }
        }
        BackoffStrategy::Linear { interval } => *interval * attempt as u32,
        BackoffStrategy::Fixed { interval } => *interval,
    }
}

/// Execute a function with retry logic
pub async fn with_retry<F, Fut, T, E>(
    strategy: &BackoffStrategy,
    max_retries: u32,
    operation: F,
) -> Result<T, E>
where
    F: Fn() -> Fut,
    Fut: std::future::Future<Output = Result<T, E>>,
    E: std::fmt::Display,
{
    let mut attempt = 0;
    loop {
        match operation().await {
            Ok(result) => return Ok(result),
            Err(e) => {
                attempt += 1;
                if attempt >= max_retries {
                    return Err(e);
                }

                let delay = calculate_delay(strategy, attempt, Duration::from_secs(0));
                sleep(delay).await;
            }
        }
    }
}

#[cfg(test)]
mod tests {
    use super::*;

    #[test]
    fn test_exponential_backoff() {
        let strategy = BackoffStrategy::Exponential {
            base: Duration::from_millis(100),
            max: Duration::from_secs(5),
        };

        assert_eq!(calculate_delay(&strategy, 0, Duration::ZERO), Duration::from_millis(100));
        assert_eq!(calculate_delay(&strategy, 1, Duration::ZERO), Duration::from_millis(200));
        assert_eq!(calculate_delay(&strategy, 2, Duration::ZERO), Duration::from_millis(400));
        assert!(calculate_delay(&strategy, 10, Duration::ZERO) <= Duration::from_secs(5));
    }

    #[test]
    fn test_linear_backoff() {
        let strategy = BackoffStrategy::Linear {
            interval: Duration::from_millis(50),
        };

        assert_eq!(calculate_delay(&strategy, 0, Duration::ZERO), Duration::from_millis(50));
        assert_eq!(calculate_delay(&strategy, 1, Duration::ZERO), Duration::from_millis(100));
        assert_eq!(calculate_delay(&strategy, 2, Duration::ZERO), Duration::from_millis(150));
```

```rust
use super::{Workflow, Step, StepType, WorkflowResult, StepResult, WorkflowStatus, StepStatus, ConditionType, BackoffStrategy};
use super::retry::{with_retry, calculate_delay};
use crate::agent::Agent;
use crate::a2a::A2AClient;
use tokio::time::{timeout, sleep};
use std::time::Duration;
use std::sync::Arc;

pub struct Scheduler {
    agent: Arc<Agent>,
    a2a_client: Arc<A2AClient>,
}

impl Scheduler {
    pub fn new(agent: Arc<Agent>, a2a_client: Arc<A2AClient>) -> Self {
        Self { agent, a2a_client }
    }

    /// Execute a workflow and return results
    pub async fn execute_workflow(&self, workflow: &Workflow) -> WorkflowResult {
        let start = std::time::Instant::now();
        let mut step_results = Vec::new();
        let mut errors = Vec::new();
        let mut current_input = serde_json::json!({});

        for step in &workflow.steps {
            match self.execute_step(step, &current_input).await {
                Ok(result) => {
                    current_input = result.output.clone().unwrap_or(serde_json::Value::Null);
                    step_results.push(result);
                }
                Err(e) => {
                    errors.push(format!("Step '{}' failed: {}", step.name, e));
                    step_results.push(StepResult {
                        step_name: step.name.clone(),
                        status: StepStatus::Failed { error: e.to_string() },
                        output: None,
                        duration: Duration::ZERO,
                        retry_count: 0,
                    });
                    break;  // Stop workflow on step failure
                }
            }

            // Handle conditional branching
            if let StepType::Conditional { condition } = &step.step_type {
                if let Some(output) = &current_input {
                    if self.evaluate_condition(condition, output) {
                        // Continue to next step (true branch)
                    } else {
                        // Skip to false branch or end
                        break;
                    }
                }
            }
        }

        let duration = start.elapsed();
        let status = if errors.is_empty() {
            WorkflowStatus::Completed
        } else {
            WorkflowStatus::Failed { failed_step: step_results.last().unwrap().step_name.clone() }
        };

        WorkflowResult {
            workflow_name: workflow.name.clone(),
            status,
            step_results,
            duration,
            errors,
        }
    }

    /// Execute individual step with timeout and retry
    async fn execute_step(&self, step: &Step, input: &serde_json::Value) -> Result<StepResult, Box<dyn std::error::Error>> {
        let start = std::time::Instant::now();

        match step.step_type {
            StepType::Sequential | StepType::Parallel => {
                // Parallel group detection (simplified)
                let future = if step.agent_id.is_some() {
                    self.execute_remote_step(step, input.clone()).boxed()
                } else {
                    self.execute_local_step(step, input.clone()).boxed()
                };

                let result = timeout(step.timeout, future).await.map_err(|_| {
                    std::io::Error::new(std::io::ErrorKind::TimedOut, "Step timeout")
                })??;

                Ok(result)
            }
            StepType::Conditional { .. } => {
                // Conditional steps don't execute, just evaluate
                Ok(StepResult {
                    step_name: step.name.clone(),
                    status: StepStatus::Completed,
                    output: None,
                    duration: start.elapsed(),
                    retry_count: 0,
                })
            }
        }
    }

    /// Execute step locally using Agent
    async fn execute_local_step(
        &self,
        step: &Step,
        input: serde_json::Value,
    ) -> Result<StepResult, Box<dyn std::error::Error>> {
        let mut retry_count = 0;

        let result = with_retry(&step.backoff, step.max_retries, || async {
            retry_count += 1;
            self.agent.run_step(&step.name, &input).await
        }).await?;

        Ok(StepResult {
            step_name: step.name.clone(),
            status: StepStatus::Completed,
            output: Some(result),
            duration: std::time::Instant::now() - std::time::Instant::now(),  // Placeholder
            retry_count,
        })
    }

    /// Execute step remotely via A2A
    async fn execute_remote_step(
        &self,
        step: &Step,
        input: serde_json::Value,
    ) -> Result<StepResult, Box<dyn std::error::Error>> {
        let agent_id = step.agent_id.as_ref().ok_or("Remote step missing agent_id")?;

        let task_id = self.a2a_client.delegate_task(
            agent_id.clone(),
            input,
        ).await?;

        // Poll for result with timeout (simplified)
        let output = self.a2a_client.poll_result(&task_id, step.timeout).await?;

        Ok(StepResult {
            step_name: step.name.clone(),
            status: StepStatus::Completed,
            output: Some(output),
            duration: std::time::Instant::now() - std::time::Instant::now(),  // Placeholder
            retry_count: 0,
        })
    }

    /// Evaluate conditional branching condition
    fn evaluate_condition(&self, condition: &ConditionType, output: &serde_json::Value) -> bool {
        match condition {
            ConditionType::JsonPath { path, expected } => {
                // Simple JSON path evaluation (placeholder)
                // In production, use proper JSON path library
                output.get(path) == Some(expected)
            }
            ConditionType::Script { expression } => {
                // Optional: simple expression eval
                // For now, always false (placeholder)
                false
            }
        }
    }
}
```
