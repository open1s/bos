# Phase 03 Plan 02: Session Persistence

## Objective

Implement agent session persistence that can serialize agent state to disk and restore it, enabling long-running conversations to survive restarts.

## Goals

1. **AgentState Serialization** — Serialize message history, context, and metadata to JSON
2. **Session Storage** — Save/load sessions to/from disk with structured directory layout
3. **Session Manager** — CRUD operations for sessions (create, read, update, delete, list)
4. **Session TTL** — Optional expiration with automatic cleanup
5. **Thread Safety** — Safe concurrent access to session store

```rust
use super::{AgentState, SessionMetadata, SessionError};
use crate::agent::agent::Message;
use std::time::{SystemTime, UNIX_EPOCH};
use flate2::{write::GzEncoder, Compression, read::GzDecoder};
use std::io::{Read, Write};

pub struct SessionSerializer;

impl SessionSerializer {
    /// Serialize AgentState to JSON bytes
    pub fn serialize(state: &AgentState) -> Result<Vec<u8>, SessionError> {
        serde_json::to_vec(state)
            .map_err(|e| SessionError::Serialization(e.to_string()))
    }

    /// Deserialize JSON bytes to AgentState
    pub fn deserialize(bytes: &[u8]) -> Result<AgentState, SessionError> {
        serde_json::from_slice(bytes)
            .map_err(|e| SessionError::Serialization(e.to_string()))
    }

    /// Create new AgentState from scratch
    pub fn new_state(agent_id: String) -> AgentState {
        let now = Self::now_timestamp();
        AgentState {
            agent_id: agent_id.clone(),
            message_log: Vec::new(),
            context: serde_json::Value::Null,
            metadata: SessionMetadata {
                created_at: now,
                updated_at: now,
                expires_at: None,
                message_count: 0,
                agent_version: "1.0.0".to_string(),
                labels: Vec::new(),
            },
        }
    }

    /// Update metadata when state changes
    pub fn update_metadata(state: &mut AgentState) {
        state.metadata.updated_at = Self::now_timestamp();
        state.metadata.message_count = state.message_log.len();
    }

    /// Get current timestamp in seconds
    fn now_timestamp() -> u64 {
        SystemTime::now()
            .duration_since(UNIX_EPOCH)
            .unwrap()
            .as_secs()
    }

    /// Check if session has expired
    pub fn is_expired(state: &AgentState) -> bool {
        if let Some(expires_at) = state.metadata.expires_at {
            let now = Self::now_timestamp();
            now >= expires_at
        } else {
            false
        }
    }

    /// Compress data (optional)
    pub fn compress(data: &[u8]) -> Result<Vec<u8>, SessionError> {
        let mut encoder = GzEncoder::new(Vec::new(), Compression::default());
        encoder.write_all(data)
            .map_err(|e| SessionError::Serialization(e.to_string()))?;
        encoder.finish()
            .map_err(|e| SessionError::Serialization(e.to_string()))
    }

    /// Decompress data
    pub fn decompress(data: &[u8]) -> Result<Vec<u8>, SessionError> {
        let mut decoder = GzDecoder::new(Vec::new());
        decoder.write_all(data)
            .map_err(|e| SessionError::Serialization(e.to_string()))?;
        decoder.finish()
            .map_err(|e| SessionError::Serialization(e.to_string()))
    }
}
```

**Implementation:**

```rust
use super::{AgentState, SessionError, SessionConfig};
use std::path::{Path, PathBuf};
use tokio::fs;

pub struct SessionStorage {
    config: SessionConfig,
}

impl SessionStorage {
    pub fn new(config: SessionConfig) -> Self {
        Self { config }
    }

    /// Ensure base directory exists
    pub async fn ensure_base_dir(&self) -> Result<(), SessionError> {
        fs::create_dir_all(&self.config.base_dir).await?;
        Ok(())
    }

    /// Get file path for a session
    fn session_path(&self, agent_id: &str) -> PathBuf {
        self.config.base_dir.join(format!("{}.json", agent_id))
    }

    /// Check if session file exists
    async fn exists(&self, agent_id: &str) -> bool {
        self.session_path(agent_id).exists()
    }

    /// Save session to disk
    pub async fn save(&self, state: &AgentState) -> Result<(), SessionError> {
        self.ensure_base_dir().await?;

        let bytes = crate::session::serializer::SessionSerializer::serialize(state)?;

        if self.config.compression_enabled {
            let compressed = crate::session::serializer::SessionSerializer::compress(&bytes)?;
            fs::write(self.session_path(&state.agent_id), compressed).await?;
        } else {
            fs::write(self.session_path(&state.agent_id), bytes).await?;
        }

        Ok(())
    }

    /// Load session from disk
    pub async fn load(&self, agent_id: &str) -> Result<AgentState, SessionError> {
        let path = self.session_path(agent_id);

        if !path.exists() {
            return Err(SessionError::NotFound(agent_id.to_string()));
        }

        let bytes = fs::read(&path).await?;

        let data = if self.config.compression_enabled {
            crate::session::serializer::SessionSerializer::decompress(&bytes)?
        } else {
            bytes
        };

        let state = crate::session::serializer::SessionSerializer::deserialize(&data)?;

        // Check expiration
        if crate::session::serializer::SessionSerializer::is_expired(&state) {
            return Err(SessionError::Expired(agent_id.to_string()));
        }

        Ok(state)
    }

    /// Delete session from disk
    pub async fn delete(&self, agent_id: &str) -> Result<(), SessionError> {
        let path = self.session_path(agent_id);

        if !path.exists() {
            return Err(SessionError::NotFound(agent_id.to_string()));
        }

        fs::remove_file(&path).await?;
        Ok(())
    }

    /// List all session files in base directory
    pub async fn list_files(&self) -> Result<Vec<PathBuf>, SessionError> {
        let mut entries = fs::read_dir(&self.config.base_dir)
            .await
            .map_err(SessionError::Io)?;

        let mut files = Vec::new();
        while let Some(entry) = entries.next_entry().await.map_err(SessionError::Io)? {
            let path = entry.path();
            if path.extension().map_or(false, |ext| ext == "json") {
                files.push(path);
            }
        }

        Ok(files)
    }
}
```

### Task 4: Session Manager (`crates/agent/src/session/manager.rs`)

**Files to create:**
- `crates/agent/src/session/manager.rs`

**Implementation:**

```rust
use super::{AgentState, SessionSummary, SessionConfig, SessionStorage, SessionError};
use super::serializer::SessionSerializer;
use std::collections::HashMap;
use std::sync::Arc;
use tokio::sync::RwLock;
use std::time::Duration;
use tokio::task::JoinHandle;

pub struct SessionManager {
    storage: SessionStorage,
    cache: Arc<RwLock<HashMap<String, AgentState>>>,
    config: SessionConfig,
    cleanup_task: Option<JoinHandle<()>>,
}

impl SessionManager {
    pub fn new(config: SessionConfig) -> Self {
        let storage = SessionStorage::new(config.clone());
        let cache = Arc::new(RwLock::new(HashMap::new()));

        let manager = Self {
            storage,
            cache,
            config,
            cleanup_task: None,
        };

        manager
    }

    /// Start background cleanup task
    pub async fn start_cleanup(&mut self, interval: Duration) {
        let cache = self.cache.clone();
        let base_dir = self.config.base_dir.clone();

        self.cleanup_task = Some(tokio::spawn(async move {
            loop {
                tokio::time::sleep(interval).await;
                Self::cleanup_expired(&cache, &base_dir).await;
            }
        }));
    }

    /// Cleanup expired sessions
    async fn cleanup_expired(
        cache: &Arc<RwLock<HashMap<String, AgentState>>>,
        base_dir: &Path,
    ) {
        let mut cache_guard = cache.write().await;
        let mut to_remove = Vec::new();

        for (agent_id, state) in cache_guard.iter() {
            if SessionSerializer::is_expired(state) {
                to_remove.push(agent_id.clone());
            }
        }

        for agent_id in to_remove {
            cache_guard.remove(&agent_id);
            // Cleanup file (simplified)
            let _ = tokio::fs::remove_file(base_dir.join(format!("{}.json", agent_id))).await;
        }
    }

    /// Create a new session
    pub async fn create(&self, agent_id: String) -> Result<AgentState, SessionError> {
        if self.exists(&agent_id).await {
            return Err(SessionError::AlreadyExists(agent_id));
        }

        let state = SessionSerializer::new_state(agent_id.clone());

        // Add to cache
        let mut cache = self.cache.write().await;
        cache.insert(agent_id.clone(), state.clone());

        // Save to disk
        self.storage.save(&state).await?;

        Ok(state)
    }

    /// Get a session (from cache or disk)
    pub async fn get(&self, agent_id: &str) -> Result<AgentState, SessionError> {
        // Check cache first
        {
            let cache = self.cache.read().await;
            if let Some(state) = cache.get(agent_id) {
                return Ok(state.clone());
            }
        }

        // Load from disk
        let state = self.storage.load(agent_id).await?;

        // Add to cache
        let mut cache = self.cache.write().await;
        cache.insert(agent_id.to_string(), state.clone());

        Ok(state)
    }

    /// Update a session
    pub async fn update(&self, agent_id: &str, mut state: AgentState) -> Result<(), SessionError> {
        if !self.exists(agent_id).await {
            return Err(SessionError::NotFound(agent_id.to_string()));
        }

        SessionSerializer::update_metadata(&mut state);

        // Update cache
        let mut cache = self.cache.write().await;
        cache.insert(agent_id.to_string(), state.clone());

        // Save to disk
        self.storage.save(&state).await?;

        Ok(())
    }

    /// Delete a session
    pub async fn delete(&self, agent_id: &str) -> Result<(), SessionError> {
        self.storage.delete(agent_id).await?;

        // Remove from cache
        let mut cache = self.cache.write().await;
        cache.remove(agent_id);

        Ok(())
    }

    /// List all sessions
    pub async fn list(&self) -> Result<Vec<SessionSummary>, SessionError> {
        let files = self.storage.list_files().await?;
        let mut summaries = Vec::new();

        for file in files {
            // Extract agent_id from filename
            if let Some(stem) = file.file_stem() {
                if let Some(agent_id) = stem.to_str() {
                    if let Ok(state) = self.storage.load(agent_id).await {
                        summaries.push(SessionSummary {
                            agent_id: agent_id.to_string(),
                            created_at: state.metadata.created_at,
                            updated_at: state.metadata.updated_at,
                            message_count: state.metadata.message_count,
                            labels: state.metadata.labels.clone(),
                            expires_at: state.metadata.expires_at,
                        });
                    }
                }
            }
        }

        Ok(summaries)
    }

    /// Check if session exists
    async fn exists(&self, agent_id: &str) -> bool {
        // Check cache first
        {
            let cache = self.cache.read().await;
            if cache.contains_key(agent_id) {
                return true;
            }
        }

        // Check disk
        self.storage.exists(agent_id).await
    }

    /// Clear cache (force reload from disk)
    pub async fn clear_cache(&self) {
        let mut cache = self.cache.write().await;
        cache.clear();
    }
}
```

```rust
use crate::session::{SessionManager, AgentState, SessionConfig, SessionError};

impl Agent {
    /// Save current agent state to session
    pub async fn save_state(&self, manager: &SessionManager) -> Result<(), SessionError> {
        let state = AgentState {
            agent_id: self.config.id.clone(),
            message_log: self.message_log.clone(),
            context: serde_json::json!({}),  // Placeholder
            metadata: crate::session::serializer::SessionSerializer::new_state(self.config.id.clone()).metadata,
        };

        manager.update(&self.config.id, state).await
    }

    /// Restore agent state from session
    pub async fn restore_state(&mut self, manager: &SessionManager) -> Result<(), SessionError> {
        let state = manager.get(&self.config.id).await?;

        self.message_log = state.message_log;

        Ok(())
    }

    /// Auto-save after each turn (optional)
    pub async fn auto_save(&self, manager: &SessionManager) {
        if let Err(e) = self.save_state(manager).await {
            eprintln!("Failed to auto-save session: {}", e);
        }
    }
}
```
