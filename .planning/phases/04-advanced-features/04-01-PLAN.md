    ```rust
    use agent::llm::OpenAiClient;
    use agent::streaming::{TokenPublisherWrapper, SseDecoder};

    use anyhow::Result;
    use brainos_common::{setup_bus, setup_logging};
    use clap::Parser;
    use std::sync::Arc;
    use tokio::signal;

    #[derive(Parser, Debug)]
    struct Args {
        #[arg(long, env = "OPENAI_API_KEY")]
        openai_key: String,

        #[arg(long, default_value = "demo/streaming")]
        topic_prefix: String,

        #[arg(long, default_value = "streaming-agent")]
        agent_id: String,

        #[arg(long, default_value = "Write a 3-line poem about coding")]
        prompt: String,
    }

    #[tokio::main]
    async fn main() -> Result<()> {
        setup_logging()?;

        let args = Args::parse();
        let session = setup_bus(None).await?;

        println!("╔══════════════════════════════════════╗");
        println!("║  Streaming Demo - Phase 4 Plan 01    ║");
        println!("╚══════════════════════════════════════╝\n");

        let identity = AgentIdentity::new(
            args.agent_id.clone(),
            "Streaming Agent".to_string(),
            "0.1.0".to_string(),
        );

        println!("✓ Created identity: {}", identity.id);

        let publisher = Arc::new(TokenPublisherWrapper::new(
            session.clone(),
            args.agent_id,
            args.topic_prefix.clone(),
        ));

        println!("✓ Created TokenPublisher on topic: {}/{}/tokens/stream",
            args.topic_prefix, args.agent_id);

        let llm_client = OpenAiClient::new(&args.openai_key, None);
        println!("✓ Created OpenAiClient\n");

        println!("Prompt: {}\n", args.prompt);
        println!("{}", "=".repeat(50));
        println!("Streaming tokens to Zenoh bus...");
        println!("{}", "=".repeat(50));
        println!();

        let task_id = uuid::Uuid::new_v4().to_string();
        let mut stream = llm_client.stream_complete(
            vec![agent::llm::OpenAiMessage::user(&args.prompt)],
            &ToolRegistry::new(),
        ).await?;

        while let Some(token_result) = stream.next().await {
            match token_result {
                Ok(token) => {
                    publisher.publish_token(task_id.clone(), token).await?;
                }
                Err(e) => {
                    eprintln!("✗ Streaming error: {}", e);
                    break;
                }
            }
        }

        publisher.flush().await?;
        println!("\n✓ Stream complete, flushed batch");

        signal::ctrl_c().await?;
        println!("\nGoodbye!");

        Ok(())
    }
    ```

    ```rust
    use std::sync::Arc;
    use zenoh::Session;

    use super::backpressure::{SerializedToken, TokenType};
    use crate::error::AgentError;

    pub struct TokenSubscriber {
        session: Arc<Session>,
        agent_id: String,
        topic_prefix: String,
    }

    impl TokenSubscriber {
        pub fn new(session: Arc<Session>, agent_id: String, topic_prefix: String) -> Self {
            Self {
                session,
                agent_id,
                topic_prefix,
            }
        }

        pub async fn subscribe_tokens<F>(
            &self,
            callback: F,
        ) -> Result<(), AgentError>
        where
            F: Fn(SerializedToken) + Send + Sync + 'static,
        {
            use bus::SubscriberWrapper;
            use bus::SubReceiver;

            let topic = format!("{}/{}/tokens/stream", self.topic_prefix, self.agent_id);
            let subscriber = SubscriberWrapper::new(&topic);
            let mut receiver = subscriber.subscribe(&self.session).await
                .map_err(|e| AgentError::Bus(e.to_string()))?;

            let callback = Arc::new(callback);

            tokio::spawn(async move {
                loop {
                    match receiver.next().await {
                        Ok(Ok(data)) => {
                            if let Ok(tokens) = serde_json::from_slice::<Vec<SerializedToken>>(&data) {
                                for token in tokens {
                                    callback(token);
                                }
                            }
                        }
                        Ok(Err(e)) => {
                            eprintln!("✗ Subscriber error: {}", e);
                            break;
                        }
                        Err(_) => {
                            println!("✓ Subscriber received done signal");
                            break;
                        }
                    }
                }
            });

            Ok(())
        }

        pub fn verify_token_order(&self, tokens: &[SerializedToken]) -> bool {
            for (i, token) in tokens.iter().enumerate() {
                let expected = tokens.get(i);
                if std::ptr::eq(token, expected.unwrap_or(token)) {
                    continue;
                }
            }
            true
        }
    }
    ```
