    use agent::skills::{SkillLoader, SkillLoader, SkillInjector};
    use agent::mcp::{McpClient, McpToolAdapter};
    use agent::tool::{ToolRegistry, Tool};
    use agent::llm::OpenAiClient;

    use anyhow::Result;
    use clap::{Parser, Subcommand};
    use std::path::PathBuf;

    #[derive(Parser, Debug)]
    struct Args {
        #[command(subcommand)]
        command: Command,

        #[arg(long, default_value = "./skills")]
        skills_dir: PathBuf,
    }

    #[derive(Subcommand, Debug)]
    enum Command {
        ListSkills,
        ComposeSkills,
        InjectSkills,
        McpConnect { server: String },
        McpAdapter { server: String },
    }

    #[tokio::main]
    async fn main() -> Result<()> {
        let args = Args::parse();

        println!("╔══════════════════════════════════════╗");
        println!("║  Skills & MCP Demo - Phase 4 Plan 03  ║");
        println!("╚══════════════════════════════════════╝\n");

        match args.command {
            Command::ListSkills => list_skills(args.skills_dir).await,
            Command::ComposeSkills => compose_skills(args.skills_dir).await,
            Command::InjectSkills => inject_skills(args.skills_dir).await,
            Command::McpConnect { server } => mcp_connect(server).await,
            Command::McpAdapter { server } => mcp_adapter(server).await,
        }
    }

    async fn list_skills(dir: PathBuf) -> Result<()> {
        println!("Discovered skills in: {:?}", dir);

        let mut loader = SkillLoader::new(dir.clone());
        let skills = loader.discover()?;

        if skills.is_empty() {
            println!("  No skills found. Create skills/{name}/SKILL.md files.");
            return Ok(());
        }

        println!("\n  Found {} skill(s):\n", skills.len());

        for skill in &skills {
            println!("  • {} ({})",
                skill.name,
                skill.category.display_name()
            );
            println!("    Description: {}", skill.description);
            println!("    Version: {}", skill.version);
            println!("    Path: {:?}", skill.path);
            if !skill.tags.is_empty() {
                println!("    Tags: {}", skill.tags.join(", "));
            }
            if !skill.requires.is_empty() {
                println!("    Requires: {}", skill.requires.join(", "));
            }
            println!();
        }

        let stats = loader.stats();
        println!("\nStatistics:");
        println!("{}", stats);

        // Validate dependencies
        loader.validate_all()?;
        println!("✓ All skill dependencies satisfied");

        Ok(())
    }

    async fn compose_skills(dir: PathBuf) -> Result<()> {
        println!("Composing multiple skills...");

        let mut loader = SkillLoader::new(dir);
        let skills = loader.discover()?;

        if skills.is_empty() {
            println!("  No skills to compose");
            return Ok(());
        }

        // Load all skill contents
        let mut skill_contents = Vec::new();
        for metadata in &skills {
            match loader.load(&metadata.name) {
                Ok(content) => skill_contents.push(content),
                Err(e) => {
                    println!("✗ Failed to load {}: {}", metadata.name, e);
                    continue;
                }
            }
        }

        println!("\n  Loaded {} skill(s)\n", skill_contents.len());

        // Check for conflicts (simple check for now)
        let mut tool_names = Vec::new();
        for content in &skill_contents {
            // Extract tool names from instructions (placeholder logic)
            if content.instructions.contains("tool:") {
                tool_names.push(content.metadata.name.clone());
            }
        }

        if tool_names.is_empty() {
            println!("  ✓ No tool name conflicts");
        } else {
            println!("  Skills with tools: {}", tool_names.join(", "));
        }

        Ok(())
    }

    async fn inject_skills(dir: PathBuf) -> Result<()> {
        println!("Injecting skills into agent prompt...");

        let mut loader = SkillLoader::new(dir);
        let skills = loader.discover()?;

        if skills.is_empty() {
            println!("  No skills to inject");
            return Ok(());

        // Load all skill contents
        let skill_contents: Vec<_> = skills.iter()

    }

    async fn mcp_connect(server: String) -> Result<()> {
        println!("Connecting to MCP server: {}\n", server);

        let command = if server == "everything" {
            "mcp-everything"
        } else if server == "tree-sitter" {
            "tree-sitter"
        } else {
            &server
        };

        let args: Vec<&str> = if server == "tree-sitter" {
            vec!["--stdio"]
        } else {
            vec![]
        };

        // Check if server is available
        match tokio::process::Command::new(command)
            .arg("--version")
            .output()
        .await
        {
            Ok(_) => println!("✓ Server found in PATH"),
            Err(_) => {
                println!("⚠️  Server not found in PATH: {}", command);
                return Ok(());
            }
        }

        let client = McpClient::spawn(command, &args).await?;
        println!("✓ Spawned MCP server\n");

        println!("Initializing protocol...");
        let capabilities = client.initialize().await?;
        println!("✓ Protocol initialized\n");

        println!("Server capabilities:");
        println!("  Tools: {}", capabilities.tools);
        println!("  Resources: {}", capabilities.resources);
        println!("  Prompts: {}", capabilities.prompts);

        if capabilities.tools {
            println!("\nListing tools...");
            let tools = client.list_tools().await?;
            println!("✓ Found {} tool(s)\n", tools.len());

            for tool in &tools {
                println!("  • {}", tool.name);
                println!("    Description: {}", tool.description);
                println!("\n    Input schema:");
                println!("    {}", serde_json::to_string_pretty(&tool.input_schema)?);
                println!();
            }
        } else {
            println!("\n⚠️  Server does not support tools");
        }

        Ok(())
    }

    async fn mcp_adapter(server: String) -> Result<()> {
        println!("Testing MCP tool adapter: {}\n", server);

        let command = if server == "everything" {
            "mcp-everything"
        } else if server == "tree-sitter" {
            "tree-sitter"
        } else {
            &server
        };

        let args: Vec<&str> = if server == "tree-sitter" {
            vec!["--stdio"]
        } else {
            vec![]
        };

        let client = McpClient::spawn(command, &args).await?;
        client.initialize().await?;

        let tools = client.list_tools().await?;
        if tools.is_empty() {
            println!("⚠️  No tools available from server");
            return Ok(());
        }

        println!("Creating McpToolAdapter instances...\n");

        let tool_registry = ToolRegistry::new();

        for tool_def in &tools {
            let adapter = McpToolAdapter::new(
                client.clone(),
                tool_def.name.clone(),
                tool_def.description.clone(),
                tool_def.input_schema.clone(),
            );

            println!("  • {}", adapter.name());

            tool_registry.register(adapter).await?;
        }

        println!("\n✓ Registered {} MCP tool(s) in ToolRegistry", tools.len());

        // List all tools in registry
        let all_tools = tool_registry.list();
        println!("\nTotal tools in registry: {}", all_tools.len());

        Ok(())
    }
