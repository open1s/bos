use serde::{Deserialize, Serialize};

#[derive(Debug, Serialize, Deserialize)]
pub struct JsonRpcRequest {
    pub jsonrpc: &'static str, // "2.0"
    pub id: u64,
    pub method: String,
    #[serde(skip_serializing_if = "Option::is_none")]
    pub params: Option<serde_json::Value>,
}

#[derive(Debug, Serialize, Deserialize)]
pub struct JsonRpcResponse {
    pub jsonrpc: &'static str,
    pub id: u64,
    #[serde(default)]
    pub result: Option<serde_json::Value>,
    #[serde(default)]
    pub error: Option<JsonRpcError>,
}

#[derive(Debug, Serialize, Deserialize)]
pub struct JsonRpcError {
    pub code: i32,
    pub message: String,
    #[serde(default)]
    pub data: Option<serde_json::Value>,
}

#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct ServerCapabilities {
    pub tools: bool,
    pub resources: bool,
    pub prompts: bool,
}

#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct ToolDefinition {
    pub name: String,
    pub description: String,
    pub input_schema: serde_json::Value,
}

// MCP error codes
pub const MCP_PARSE_ERROR: i32 = -32700;
pub const MCP_INVALID_REQUEST: i32 = -32600;
pub const MCP_METHOD_NOT_FOUND: i32 = -32601;
pub const MCP_INVALID_PARAMS: i32 = -32602;
pub const MCP_INTERNAL_ERROR: i32 = -32603;
pub const MCP_RESOURCE_NOT_FOUND: i32 = -32001;
pub const MCP_TOOL_NOT_FOUND: i32 = -32002;

impl JsonRpcError {
    pub fn new(code: i32, message: &str) -> Self {
        Self {
            code,
            message: message.to_string(),
            data: None,
        }
    }

    pub fn method_not_found(method: &str) -> Self {
        Self::new(MCP_METHOD_NOT_FOUND, &format!("Method not found: {}", method))
    }

    pub fn invalid_params(msg: &str) -> Self {
        Self::new(MCP_INVALID_PARAMS, msg)
    }

    pub fn parse_error(msg: &str) -> Self {
        Self::new(MCP_PARSE_ERROR, msg)
    }
}

use std::process::Stdio;
use tokio::io::{AsyncBufReadExt, AsyncWriteExt, BufReader};
use tokio::process::{Child, ChildStderr, ChildStdin, ChildStdout};

use super::McpError;

pub struct StdioTransport {
    child: Child,
    stdin: ChildStdin,
    stdout: BufReader<ChildStdout>,
    stderr: ChildStderr,
}

impl StdioTransport {
    pub async fn spawn(command: &str, args: &[&str]) -> Result<Self, McpError> {
        let mut child = tokio::process::Command::new(command)
            .args(args)
            .stdin(Stdio::piped())
            .stdout(Stdio::piped())
            .stderr(Stdio::piped())
            .kill_on_drop(true) // Prevent zombies
            .spawn()
            .map_err(|e| McpError::Transport(e.to_string()))?;

        let stdin = child.stdin.take().ok_or(McpError::Transport("no stdin".into()))?;
        let stdout = child.stdout.take().ok_or(McpError::Transport("no stdout".into()))?;
        let stderr = child.stderr.take().ok_or(McpError::Transport("no stderr".into()))?;

        Ok(Self {
            child,
            stdin,
            stdout: BufReader::new(stdout),
            stderr,
        })
    }

    pub async fn send(&mut self, msg: &[u8]) -> Result<(), McpError> {
        self.stdin.write_all(msg).await
            .map_err(|e| McpError::Transport(e.to_string()))?;
        self.stdin.write_all(b"\n").await // Newline-delimited JSON
            .map_err(|e| McpError::Transport(e.to_string()))?;
        self.stdin.flush().await
            .map_err(|e| McpError::Transport(e.to_string()))?;
        Ok(())
    }

    pub async fn recv_line(&mut self) -> Result<String, McpError> {
        let mut line = String::new();
        self.stdout.read_line(&mut line).await
            .map_err(|e| McpError::Transport(e.to_string()))?;
        Ok(line.trim().to_string())
    }

    pub async fn recv_until_close(&mut self) -> Result<String, McpError> {
        let mut output = String::new();
        let mut buf = String::new();
        while match self.stdout.read_line(&mut buf).await {
            Ok(0) => false,
            Ok(_) => {
                output.push_str(&buf);
                buf.clear();
                true
            }
            Err(e) => return Err(McpError::Transport(e.to_string())),
        } {}
        Ok(output)
    }

    pub async fn take_stderr(&mut self) -> String {
        let mut output = String::new();
        let mut buf = String::new();
        while match self.stderr.read_line(&mut buf).await {
            Ok(0) => false,
            Ok(_) => {
                output.push_str(&buf);
                buf.clear();
                true
            }
            Err(_) => false,
        } {}
        output
    }

    pub async fn shutdown(mut self) -> Result<(), McpError> {
        // Try graceful shutdown first
        if let Err(e) = self.child.wait().await {
            // If graceful fails, kill
            let _ = self.child.kill().await;
            return Err(McpError::Transport(e.to_string()));
        }
        Ok(())
    }

    pub fn is_running(&self) -> bool {
        match self.child.try_wait() {
            Ok(None) => true,
            _ => false,
        }
    }
}

use std::sync::Arc;
use tokio::sync::Mutex;

use super::{McpError, StdioTransport, JsonRpcRequest, JsonRpcResponse, JsonRpcError, ServerCapabilities, ToolDefinition};

pub struct McpClient {
    transport: Arc<Mutex<StdioTransport>>,
    request_id: std::sync::atomic::AtomicU64,
}

impl McpClient {
    pub async fn spawn(command: &str, args: &[&str]) -> Result<Self, McpError> {
        let transport = StdioTransport::spawn(command, args).await?;
        Ok(Self {
            transport: Arc::new(Mutex::new(transport)),
            request_id: std::sync::atomic::AtomicU64::new(1),
        })
    }

    pub async fn initialize(&self) -> Result<ServerCapabilities, McpError> {
        let resp = self.call("initialize", Some(serde_json::json!({
            "protocolVersion": "2025-03-26",
            "capabilities": {
                "roots": { "listChanged": true },
                "sampling": {}
            },
            "clientInfo": {
                "name": "brainos-agent",
                "version": "1.0.0"
            }
        }))).await?;

        let capabilities = resp.result
            .and_then(|v| v.get("capabilities").cloned())
            .unwrap_or(serde_json::json!({}));
        
        Ok(ServerCapabilities {
            tools: capabilities.get("tools").and_then(|v| v.as_bool()).unwrap_or(false),
            resources: capabilities.get("resources").and_then(|v| v.as_bool()).unwrap_or(false),
            prompts: capabilities.get("prompts").and_then(|v| v.as_bool()).unwrap_or(false),
        })
    }

    pub async fn list_tools(&self) -> Result<Vec<ToolDefinition>, McpError> {
        let resp = self.call("tools/list", None).await?;

        let tools = resp.result
            .and_then(|v| v.get("tools").cloned())
            .unwrap_or(serde_json::Value::Array(vec![]));

        let tools_arr = tools.as_array()
            .ok_or_else(|| McpError::Protocol("Expected tools array".to_string()))?;

        tools_arr.iter()
            .map(|t| {
                Ok(ToolDefinition {
                    name: t.get("name").and_then(|v| v.as_str())
                        .ok_or_else(|| McpError::Protocol("Missing tool name".to_string()))?
                        .to_string(),
                    description: t.get("description").and_then(|v| v.as_str())
                        .unwrap_or("")
                        .to_string(),
                    input_schema: t.get("inputSchema")
                        .cloned()
                        .unwrap_or(serde_json::json!({})),
                })
            })
            .collect()
    }

    pub async fn call_tool(&self, name: &str, args: serde_json::Value) -> Result<serde_json::Value, McpError> {
        let resp = self.call("tools/call", Some(serde_json::json!({
            "name": name,
            "arguments": args
        }))).await?;

        // MCP tool response: { "content": [{ "type": "text", "text": "..." }] }
        let content = resp.result
            .and_then(|v| v.get("content").cloned())
            .unwrap_or(serde_json::Value::Array(vec![]));

        // Extract text from first content block
        if let Some(arr) = content.as_array() {
            if let Some(first) = arr.first() {
                if let Some(text) = first.get("text") {
                    return Ok(text.clone());
                }
            }
        }

        Ok(content)
    }

    pub async fn shutdown(mut self) -> Result<(), McpError> {
        // Send shutdown notification
        let _ = self.notify("shutdown", None).await;
        
        // Wait for process to exit
        let mut transport = self.transport.lock().await;
        tokio::time::sleep(std::time::Duration::from_secs(5)).await;
        
        // Force kill if still running
        if transport.is_running() {
            let _ = transport.child.kill().await;
        }
        
        Ok(())
    }

    async fn call(&self, method: &str, params: Option<serde_json::Value>) -> Result<JsonRpcResponse, McpError> {
        let id = self.request_id.fetch_add(1, std::sync::atomic::Ordering::SeqCst);
        
        let req = JsonRpcRequest {
            jsonrpc: "2.0",
            id,
            method: method.to_string(),
            params,
        };

        let json = serde_json::to_vec(&req)
            .map_err(|e| McpError::Protocol(e.to_string()))?;

        let mut transport = self.transport.lock().await;
        transport.send(&json).await?;

        let line = transport.recv_line().await?;
        
        if line.is_empty() {
            return Err(McpError::Protocol("Empty response".to_string()));
        }

        let resp: JsonRpcResponse = serde_json::from_str(&line)
            .map_err(|e| McpError::Protocol(e.to_string()))?;

        if let Some(error) = resp.error {
            return Err(McpError::JsonRpc(error));
        }

        Ok(resp)
    }

    async fn notify(&self, method: &str, params: Option<serde_json::Value>) -> Result<(), McpError> {
        let req = serde_json::json!({
            "jsonrpc": "2.0",
            "method": method,
            "params": params
        });

        let json = serde_json::to_vec(&req)
            .map_err(|e| McpError::Protocol(e.to_string()))?;

        let mut transport = self.transport.lock().await;
        transport.send(&json).await
    }
}

use async_trait::async_trait;
use std::sync::Arc;
use tokio::sync::Mutex;

use crate::tools::{Tool, ToolDescription, ToolError};
use super::{McpClient, ToolDefinition};

pub struct McpToolAdapter {
    client: Arc<McpClient>,
    tool_name: String,
    description: String,
    input_schema: serde_json::Value,
}

impl McpToolAdapter {
    pub fn new(
        client: Arc<McpClient>,
        name: String,
        description: String,
        input_schema: serde_json::Value,
    ) -> Self {
        Self {
            client,
            tool_name: name,
            description,
            input_schema,
        }
    }

    pub fn from_definition(client: Arc<McpClient>, definition: ToolDefinition) -> Self {
        Self {
            client,
            tool_name: definition.name,
            description: definition.description,
            input_schema: definition.input_schema,
        }
    }
}

#[async_trait]
impl Tool for McpToolAdapter {
    fn name(&self) -> &str {
        &self.tool_name
    }

    fn description(&self) -> ToolDescription {
        ToolDescription {
            short: self.description.clone(),
            parameters: self.input_schema.to_string(),
        }
    }

    fn json_schema(&self) -> serde_json::Value {
        self.input_schema.clone()
    }

    async fn execute(&self, args: serde_json::Value) -> Result<serde_json::Value, ToolError> {
        self.client
            .call_tool(&self.tool_name, args)
            .await
            .map_err(|e| ToolError::ExecutionFailed(e.to_string()))
    }
}
