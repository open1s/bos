```rust
#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct Task {
    pub task_id: String,
    pub context_id: Option<String>,
    pub state: TaskState,
    pub created_at: u64,
    pub updated_at: u64,
    pub input: serde_json::Value,
    pub output: Option<serde_json::Value>,
    pub error: Option<String>,
}

#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize)]
pub enum TaskState {
    Submitted,
    Working,
    InputRequired,
    Completed,
    Failed,
    Canceled,
}

impl TaskState {
    pub fn is_terminal(self) -> bool {
        matches!(self, Self::Completed | Self::Failed | Self::Canceled)
    }
    pub fn can_transition_to(self, next: TaskState) -> bool {
        match (self, next) {
            (Self::Submitted, Self::Working) => true,
            (Self::Submitted, Self::InputRequired) => true,
            (Self::Submitted, Self::Failed) => true,
            (Self::Submitted, Self::Canceled) => true,
            (Self::Working, Self::Completed) => true,
            (Self::Working, Self::Failed) => true,
            (Self::Working, Self::InputRequired) => true,
            (Self::Working, Self::Canceled) => true,
            (Self::InputRequired, Self::Working) => true,
            (Self::InputRequired, Self::Canceled) => true,
            _ => false,
        }
    }
}
```

```rust
use std::collections::HashMap;
use std::time::{Duration, Instant};
use tokio::sync::RwLock;

#[derive(Debug, Clone)]
pub struct ProcessedResult {
    pub task_id: String,
    pub state: super::TaskState,
    pub result: Option<serde_json::Value>,
    pub timestamp: u64,
}

pub struct IdempotencyStore {
    processed: RwLock<HashMap<String, CacheEntry>>,
    default_ttl: Duration,
}

struct CacheEntry {
    result: ProcessedResult,
    expires_at: Instant,
}

impl IdempotencyStore {
    pub fn new() -> Self {
        Self {
            processed: RwLock::new(HashMap::new()),
            default_ttl: Duration::from_secs(300), // 5 min
        }
    }

    pub fn with_ttl(ttl: Duration) -> Self {
        Self {
            processed: RwLock::new(HashMap::new()),
            default_ttl: ttl,
        }
    }

    pub async fn check(&self, key: &str) -> Option<ProcessedResult> {
        let store = self.processed.read().await;
        store.get(key).and_then(|entry| {
            if entry.expires_at > Instant::now() {
                Some(entry.result.clone())
            } else {
                None
            }
        })
    }

    pub async fn record(&self, key: String, result: ProcessedResult) {
        let mut store = self.processed.write().await;
        store.insert(key, CacheEntry {
            result,
            expires_at: Instant::now() + self.default_ttl,
        });
    }

    pub async fn cleanup(&self) {
        let mut store = self.processed.write().await;
        let now = Instant::now();
        store.retain(|_, entry| entry.expires_at > now);
    }

    pub async fn remove(&self, key: &str) {
        let mut store = self.processed.write().await;
        store.remove(key);
    }

    pub async fn size(&self) -> usize {
        let store = self.processed.read().await;
        store.len()
    }
}

impl Default for IdempotencyStore {
    fn default() -> Self {
        Self::new()
    }
}

#[cfg(test)]
mod tests {
    use super::*;

    #[tokio::test]
    async fn test_idempotency_store_roundtrip() {
        let store = IdempotencyStore::new();
        let key = "test-key";
        let result = ProcessedResult {
            task_id: "task-123".to_string(),
            state: super::TaskState::Completed,
            result: Some(serde_json::json!({"output": "done"})),
            timestamp: 1000,
        };

        store.record(key.to_string(), result.clone()).await;
        let found = store.check(key).await;

        assert!(found.is_some());
        assert_eq!(found.unwrap().task_id, "task-123");
    }

    #[tokio::test]
    async fn test_idempotency_expiry() {
        let store = IdempotencyStore::with_ttl(Duration::from_millis(10));
        let key = "expire-key";
        let result = ProcessedResult {
            task_id: "task-456".to_string(),
            state: super::TaskState::Completed,
            result: None,
            timestamp: 2000,
        };

        store.record(key.to_string(), result).await;
        tokio::time::sleep(Duration::from_millis(20)).await;
        let found = store.check(key).await;

        assert!(found.is_none());
    }
}
```

```rust
use std::sync::Arc;
use std::time::Duration;
use zenoh::Session;

use super::{A2AMessage, A2AContent, AgentIdentity, Task, TaskState, IdempotencyStore, AgentCard, A2ADiscovery};

pub struct A2AClient {
    session: Arc<Session>,
    identity: AgentIdentity,
    idempotency: IdempotencyStore,
    timeout: Duration,
}

impl A2AClient {
    pub fn new(session: Arc<Session>, identity: AgentIdentity) -> Self {
        Self {
            session,
            identity,
            idempotency: IdempotencyStore::new(),
            timeout: Duration::from_secs(60),
        }
    }

    pub async fn delegate_task(
        &self,
        recipient: &AgentIdentity,
        mut task: Task,
    ) -> Result<Task, crate::error::AgentError> {
        // Check idempotency
        let key = format!("{}:{}", task.task_id, task.created_at);
        if let Some(cached) = self.idempotency.check(&key).await {
            return Ok(Task {
                state: cached.state,
                output: cached.result,
                ..task
            });
        }

        // Build A2A message
        let message = A2AMessage {
            message_id: uuid::Uuid::new_v4().to_string(),
            task_id: task.task_id.clone(),
            context_id: task.context_id.clone(),
            idempotency_key: key.clone(),
            timestamp: std::time::SystemTime::now()
                .duration_since(std::time::UNIX_EPOCH)
                .unwrap()
                .as_millis() as u64,
            sender: self.identity.clone(),
            recipient: recipient.clone(),
            content: A2AContent::TaskRequest { task: task.clone() },
        };

        // Publish to recipient's incoming tasks
        let topic = format!("agent/{}/tasks/incoming", recipient.id);
        let publisher = self.session.declare_publisher(&topic).await
            .map_err(|e| crate::error::AgentError::Bus(e.to_string()))?;
        
        let data = serde_json::to_vec(&message)
            .map_err(|e| crate::error::AgentError::Config(e.to_string()))?;
        publisher.put(data).await
            .map_err(|e| crate::error::AgentError::Bus(e.to_string()))?;

        // Poll for result
        let result = self.poll_task_result(&task.task_id, recipient).await?;

        // Record in idempotency store
        self.idempotency.record(key, crate::a2a::idempotency::ProcessedResult {
            task_id: task.task_id.clone(),
            state: result.state,
            result: result.output.clone(),
            timestamp: std::time::SystemTime::now()
                .duration_since(std::time::UNIX_EPOCH)
                .unwrap()
                .as_millis() as u64,
        }).await;

        Ok(result)
    }

    pub async fn poll_task_result(&self, task_id: &str, recipient: &AgentIdentity) -> Result<Task, crate::error::AgentError> {
        let topic = format!("agent/{}/tasks/{}/status", recipient.id, task_id);
        
        let subscriber = self.session.subscribe(&topic).await
            .map_err(|e| crate::error::AgentError::Bus(e.to_string()))?;

        let mut receiver = subscriber.recv_async();
        let timeout = tokio::time::timeout(self.timeout, async {
            while let Ok(sample) = receiver.recv_async().await {
                if let Ok(msg) = serde_json::from_slice::<A2AMessage>(&sample.value.to_bytes()) {
                    if let A2AContent::TaskResponse { task } = msg.content {
                        return task;
                    }
                }
            }
            Task {
                task_id: task_id.to_string(),
                context_id: None,
                state: TaskState::Failed,
                created_at: 0,
                updated_at: 0,
                input: serde_json::Value::Null,
                output: None,
                error: Some("Timeout waiting for result".to_string()),
            }
        }).await;

        timeout.map_err(|_| crate::error::AgentError::Timeout("Task polling timed out".to_string()))
    }

    pub async fn get_discovery(&self) -> A2ADiscovery {
        A2ADiscovery::new(self.session.clone())
    }
}
```
