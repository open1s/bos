  ```rust
  use std::time::{Duration, Instant};
  use crate::llm::StreamToken;

  #[derive(Debug, Clone)]
  pub struct TokenBatch {
      pub tokens: Vec<SerializedToken>,
      pub token_count: usize,
      pub created_at: Instant,
  }

  #[derive(Debug, Clone, serde::Serialize, serde::Deserialize)]
  pub struct SerializedToken {
      pub task_id: String,
      pub token_type: TokenType,
      pub content: String,
  }

  #[derive(Debug, Clone, serde::Serialize, serde::Deserialize)]
  pub enum TokenType {
      Text,
      ToolCall,
      Done,
  }

  impl TokenBatch {
      pub fn new() -> Self {
          Self {
              tokens: Vec::new(),
              token_count: 0,
              created_at: Instant::now(),
          }
      }

      pub fn is_full(&self, max_size: usize, max_tokens: usize) -> bool {
          self.tokens.len() >= max_size || self.token_count >= max_tokens
      }

      pub fn should_flush(&self, max_age: Duration) -> bool {
          self.created_at.elapsed() >= max_age
      }

      pub fn add(&mut self, token: SerializedToken) {
          self.token_count += 1;
          self.tokens.push(token);
      }

      pub fn clear(&mut self) {
          self.tokens.clear();
          self.token_count = 0;
          self.created_at = Instant::now();
      }

      pub fn to_bytes(&self) -> Result<Vec<u8>, serde_json::Error> {
          serde_json::to_vec(self)
      }
  }
  ```

  2. **RateLimiter** — Token bucket algorithm for rate limiting:
  ```rust
  pub struct RateLimiter {
      tokens_per_second: f64,
      last_check: Instant,
      tokens_available: f64,
  }

  impl RateLimiter {
      pub fn new(tokens_per_second: f64) -> Self {
          Self {
              tokens_per_second,
              last_check: Instant::now(),
              tokens_available: tokens_per_second,
          }
      }

      pub async fn try_acquire(&mut self) -> bool {
          let now = Instant::now();
          let elapsed = now.duration_since(self.last_check);
          self.tokens_available += elapsed.as_secs_f64() * self.tokens_per_second;
          self.tokens_available = self.tokens_available.min(self.tokens_per_second * 2.0); // Cap at 2x burst
          self.last_check = now;

          if self.tokens_available >= 1.0 {
              self.tokens_available -= 1.0;
              true
          } else {
              tokio::time::sleep(Duration::from_millis(
                  ((1.0 - self.tokens_available) * 1000.0 / self.tokens_per_second) as u64
              )).await;
              false
          }
      }

      pub fn reset(&mut self) {
          self.tokens_available = self.tokens_per_second;
          self.last_check = Instant::now();
      }
  }
  ```

  3. **BackpressureController** — Monitors bus health and adjusts rate:
  ```rust
  pub struct BackpressureController {
      rate_limiter: RateLimiter,
      max_batch_size: usize,
      max_batch_tokens: usize,
      batch_timeout: Duration,
      backpressure_threshold: f64, // 0.0-1.0
      current_load: f64,
  }

  impl BackpressureController {
      pub fn new(
          tokens_per_second: f64,
          max_batch_size: usize,
          max_batch_tokens: usize,
          batch_timeout: Duration,
      ) -> Self {
          Self {
              rate_limiter: RateLimiter::new(tokens_per_second),
              max_batch_size,
              max_batch_tokens,
              batch_timeout,
              backpressure_threshold: 0.8,
              current_load: 0.0,
          }
      }

      pub async fn should_publish(&mut self) -> bool {
          self.rate_limiter.try_acquire().await
      }

      pub fn is_batch_ready(&self, batch: &TokenBatch) -> bool {
          batch.is_full(self.max_batch_size, self.max_batch_tokens)
              || batch.should_flush(self.batch_timeout)
      }

      pub fn report_bus_load(&mut self, load: f64) {
          self.current_load = self.current_load * 0.9 + load * 0.1; // EMA

          if self.current_load > self.backpressure_threshold {
              // Reduce rate by 50% when overloaded
              let new_rate = self.rate_limiter.tokens_per_second * 0.5;
              self.rate_limiter = RateLimiter::new(new_rate);
          } else if self.current_load < 0.5 {
              // Restore rate when load is low
              self.rate_limiter = RateLimiter::new(self.rate_limiter.tokens_per_second * 1.5);
              let target_rate = self.rate_limiter.tokens_per_second.min(100.0); // Cap at 100 tokens/sec
              if self.rate_limiter.tokens_per_second < target_rate {
                  self.rate_limiter = RateLimiter::new(target_rate);
              }
          }
      }

      pub fn current_rate(&self) -> f64 {
          self.rate_limiter.tokens_per_second
      }

      pub fn get_batch_config(&self) -> (usize, usize, Duration) {
          (self.max_batch_size, self.max_batch_tokens, self.batch_timeout)
      }
  }
  ```

  ```rust
  use std::sync::Arc;
  use tokio::sync::Mutex;
  use zenoh::Session;

  use super::backpressure::{
      BackpressureController, TokenBatch, SerializedToken, TokenType,
  };
  use crate::llm::StreamToken;

  pub struct PublisherWrapper {
      session: Arc<Session>,
      backpressure: Mutex<BackpressureController>,
      batch: Mutex<TokenBatch>,
      topic_prefix: String,
      agent_id: String,
  }

  impl PublisherWrapper {
      pub fn new(
          session: Arc<Session>,
          agent_id: String,
          topic_prefix: String,
      ) -> Self {
          // Default config: 10 tokens/batch, 50ms timeout, 100 tokens/sec rate
          let backpressure = Mutex::new(BackpressureController::new(
              100.0, // tokens_per_second
              10,    // max_batch_size
              50,    // max_batch_tokens
              Duration::from_millis(50), // batch_timeout
          ));

          Self {
              session,
              backpressure,
              batch: Mutex::new(TokenBatch::new()),
              topic_prefix,
              agent_id,
          }
      }

      pub async fn publish_token(&self, task_id: String, token: StreamToken) -> Result<(), crate::error::AgentError> {
          // Serialize token
          let serialized = match token {
              StreamToken::Text(content) => SerializedToken {
                  task_id: task_id.clone(),
                  token_type: TokenType::Text,
                  content,
              },
              StreamToken::ToolCall { name, args } => {
                  let payload = serde_json::to_string(&(name, args)).unwrap_or_default();
                  SerializedToken {
                      task_id: task_id.clone(),
                      token_type: TokenType::ToolCall,
                      content: payload,
                  }
              }
              StreamToken::Done => SerializedToken {
                  task_id: task_id.clone(),
                  token_type: TokenType::Done,
                  content: String::new(),
              },
          };

          // Check rate limit
          {
              let mut bp = self.backpressure.lock().await;
              if !bp.should_publish().await {
                  drop(bp);
                  tokio::time::sleep(Duration::from_millis(10)).await;
                  return self.publish_token(task_id, token).await;
              }
          }

          // Add to batch
          {
              let mut batch = self.batch.lock().await;
              batch.add(serialized);

              let bp = self.backpressure.lock().await;
              if bp.is_batch_ready(&batch) {
                  drop(bp);
                  Self::flush_batch(self, batch).await?;
              }
          }

          Ok(())
      }

      pub async fn flush(&self) -> Result<(), crate::error::AgentError> {
          let batch = self.batch.lock().await;
          if !batch.tokens.is_empty() {
              Self::flush_batch(self, batch).await?;
          }
          Ok(())
      }

      async fn flush_batch(
          &self,
          batch: tokio::sync::MutexGuard<'_, TokenBatch>,
      ) -> Result<(), crate::error::AgentError> {
          let topic = format!("{}/{}/tokens/stream", self.topic_prefix, self.agent_id);
          let bytes = batch.to_bytes()
              .map_err(|e| crate::error::AgentError::Config(e.to_string()))?;

          let publisher = self.session.declare_publisher(&topic).await
              .map_err(|e| crate::error::AgentError::Bus(e.to_string()))?;

          publisher.put(bytes).await
              .map_err(|e| crate::error::AgentError::Bus(e.to_string()))?;

          // Clear batch via mutable access
          drop(batch);
          let mut batch = self.batch.lock().await;
          batch.clear();

          Ok(())
      }

      pub async fn report_bus_load(&self, load: f64) {
          let mut bp = self.backpressure.lock().await;
          bp.report_bus_load(load);
      }

      pub fn get_rate(&self) -> f64 {
          let bp = self.backpressure.blocking_lock();
          bp.current_rate()
      }

      pub async fn with_config<F, R>(&self, f: F) -> R
      where
          F: FnOnce(&BackpressureController) -> R,
      {
          let bp = self.backpressure.lock().await;
          f(&bp)
      }
  }

  // Convenience type for direct token publishing
  pub struct TokenPublisher {
      wrapper: Arc<PublisherWrapper>,
  }

  impl TokenPublisher {
      pub fn new(wrapper: Arc<PublisherWrapper>) -> Self {
          Self { wrapper }
      }

      pub async fn publish(&self, task_id: String, token: StreamToken) -> Result<(), crate::error::AgentError> {
          self.wrapper.publish_token(task_id, token).await
      }
  }
  ```
